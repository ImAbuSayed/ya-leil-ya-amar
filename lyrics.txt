Song
يـا ليل يـا قمر (Ya Leil Ya Amar)
Lyrics
[Verse 1]
يا ليل يـا قمر .. والناس نايمـة
قلبـي اتحرّق نـار .. وعينـك ساحرتنـا
لو تدّيني إشارة .. أرقص لـ حُبّك فـ لحظة
إيدك فـ إيـدي .. والسّمـع كلامـك
[Chorus]
يـا حبيبي دُنـيا .. رقصـة ولا أغنـى
عـ البـال تيك توك .. وقلبـك يِغـنّي لـي
يـا حُبّ اللي حيّرنـي .. أنا مِشتاقة لـ خطوتك
إوعـى تِسيب إيدك .. واحضنـي لـ آخـر دقيقـة
[Verse 2]
النهارـدة لـيـا .. يوم مِـن الأحلام
ضَحكتـك سِحـر .. خدّك وردة فـ الـيام
قول لي “أنـا لـيـك” .. وأوعـى تِنسـى المواعيد
رقصـة بـلا موسيقى .. بس إحسـاسـك كفايـة
[Bridge]
لـمّـا بتِغمض عينيـك .. بتِفتكـر إيه فـ بالـك؟
هلّـا غنـي لـي وِرد .. وخدّنـي عـ السحابـة
إنتَ جِنّـي ولا إنسـي؟ .. مِين علّـمـك الغرام؟
أنا مالي غيرك .. حُـبّـك نار وبـردة
[Outro]
يـا ليل يـا قمر .. خلّي الضّـو يِتلـه
إحنا الاتنين غنـوة .. مكتوبة منـذ زمان
إوعـى تفـرق إيدنـا .. لـ آخر العُـمر خلّيـها
رقصـة بـلا نهايـة .. وحُـبّـنـا لـهيبـه
